module.exports = async function handler(req, res) { try { // CORS headers res.setHeader("Access-Control-Allow-Origin", "*"); res.setHeader("Access-Control-Allow-Methods", "POST, OPTIONS"); res.setHeader("Access-Control-Allow-Headers", "Content-Type"); if (req.method === "OPTIONS") { return res.status(200).end(); } if (req.method !== "POST") { return res.status(405).json({ error: "Method not allowed" }); } const apiKey = process.env.ANTHROPIC_API_KEY; if (!apiKey) { // Return error frame const errorHtml = ` `; return res.setHeader('Content-Type', 'text/html').status(200).send(errorHtml); } // Get current date const today = new Date(); const month = today.toLocaleString('en-US', { month: 'long' }); const day = today.getDate(); // Call Anthropic API const response = await fetch("https://api.anthropic.com/v1/messages", { method: "POST", headers: { "x-api-key": apiKey, "anthropic-version": "2023-06-01", "Content-Type": "application/json", }, body: JSON.stringify({ model: "claude-sonnet-4-20250514", max_tokens: 1024, messages: [{ role: "user", content: `Give me 3 interesting historical events that happened on ${month} ${day}. Keep each event brief (one line). Format as: üìç YEAR: Brief description Make them diverse and fascinating. Respond with ONLY the formatted events.` }] }), }); // Handle rate limits if (response.status === 429 || response.status === 400 || response.status === 402) { const limitHtml = ` `; return res.setHeader('Content-Type', 'text/html').status(200).send(limitHtml); } if (!response.ok) { throw new Error('API request failed'); } const data = await response.json(); const content = data.content?.[0]?.text?.trim(); if (!content) { throw new Error('No content generated'); } // Generate image URL with the history text const encodedText = encodeURIComponent(content.substring(0, 200)); const title = encodeURIComponent(`${month} ${day} in History`); const imageUrl = `https://og-image.vercel.app/${title}.png?theme=dark&md=1&fontSize=50px&images=https%3A%2F%2Fassets.vercel.com%2Fimage%2Fupload%2Ffront%2Fassets%2Fdesign%2Fvercel-triangle-white.svg`; // Return frame with history const frameHtml = ` `; res.setHeader('Content-Type', 'text/html'); res.status(200).send(frameHtml); } catch (err) { console.error("Frame Error:", err); const errorHtml = ` `; res.setHeader('Content-Type', 'text/html'); res.status(200).send(errorHtml); } };
